19 bool Scheduler::start(IdleTaskControlBlock& idle)
20 {
21  assert((CortexM::getType() == CortexM::Type::M4) ||(CortexM::getType() == CortexM::Type::M7)); // only Cortex-M4 and M7 are officially supported so far
22  assert(!s_isStarted);
23  s_isStarted = true;
24 
25  uint32_t ratio = std::chrono::duration_cast<duration>(
26  std::chrono::duration<int64_t>(1)).count(); // get ratio from timeout clock (system ticks to 1Hz)
27 
28  auto coreClock = getCoreClock();
29  s_idle = &idle;
30 
31  CortexM::preemptBits(kPreemptionBits);
32  CortexM::setPriority(CortexM::SystemIrq::ServiceCall, kServiceCallPriority); // service call is at system preemption and highest sub priority
33  CortexM::setPriority(CortexM::SystemIrq::PendSV, kPendSvPriority); // pending service is at lowest possible priority
34  CortexM::setPriority(CortexM::SystemIrq::Systick, kSystickPriority); // systick handler is at system preemption with lowest sub priority
35  CortexM::setIsrHandler(CortexM::SystemIrq::Systick, ::SysTick_Handler);
36  CortexM::setIsrHandler(CortexM::SystemIrq::PendSV, ::PendSV_Handler);
37  CortexM::setIsrHandler(CortexM::SystemIrq::ServiceCall, ::SVC_Handler);
38 
39  assert(coreClock % ratio == 0u); // for exact time clock the core clock divided by ratio should not leave a remainder
40  CortexM::enableSystick(coreClock / ratio);
41 
42  Hooks::starting(idle, coreClock, [](Callback<void(const TaskControlBlock&)> callback)
43  {
44  for(const auto& task : s_allTasks)
45  callback(task);
46  });
47 
48  CortexM::setPsp(CortexM::getMsp());
49  CortexM::setControl(0b10); // TODO : make flags for "control" operations !
50  CortexM::setMsp(CortexM::mspAtReset());
51 
52  return doSwitch();
53 }

55 bool Scheduler::doSwitch()
56 {
57  assert(s_isStarted);
58  assert((s_currentTask != nullptr) || (s_criticalSection == false)); // can't have no current task in critical section
59 
60  if(s_criticalSection)
61  {
62  s_mayNeedSwitch = true;
63  return false;
64  }
65 
66  const auto current = s_currentTask;
67 
68  if (s_nextTask != nullptr)
69  {
70  assert(s_currentTask != s_nextTask);
71 
72  s_ready.insertWhen(TaskControlBlock::priorityIsLower, *s_nextTask);
73  s_nextTask = nullptr;
74  }
75 
76  if (s_currentTask != nullptr)
77  {
78  s_ready.insertWhen(TaskControlBlock::priorityIsLower, *s_currentTask);
79  s_currentTask = nullptr;
80  }
81 
82  if (s_ready.empty())
83  {
84  CortexM::triggerPendSv();
85  return true;
86  }
87  else
88  {
89  s_nextTask = &s_ready.front();
90  s_ready.pop_front();
91 
92  if (s_nextTask == current)
93  {
94  s_currentTask = s_nextTask;
95  s_nextTask = nullptr;
96  return false;
97  }
98  else
99  {
100  CortexM::triggerPendSv();
101  return true;
102  }

117 void Scheduler::SystickHandler()
118 {
119  Hooks::enterSystick();
120  s_ticks+=duration(1); // this is correct and "atomic" because nothing that has preemptive level above system should use it
121 
122  bool dirty = false;
123 
124 
125  while(!s_timeouts.empty() && s_timeouts.front().m_waitUntil.value() <= s_ticks)
126  {
127  auto& task = s_timeouts.front();
128  s_timeouts.pop_front();
129  task.m_waitUntil = std::nullopt;
130 
131  if(task.m_waiting != nullptr)
132  {
133  task.m_waiting->removeWaiting(task);
134  task.setReturnValue(static_cast<uint32_t>(std::cv_status::timeout)); // notify timeout to thread (write value to its R0 frame)
135  }
136 
137  s_ready.insertWhen(TaskControlBlock::priorityIsLower, task);
138  Hooks::taskReady(task);
139  dirty = true;
140  }
141 
142  if(dirty)
143  Hooks::exitSystick(doSwitch());
144  else
145  Hooks::exitSystick(false);
146 }

148 uint64_t Scheduler::pendSvHandler(uint32_t* psp)
149 {
150  Hooks::enterPendSv();
151  CortexM::clearPendSv();
152 
153  //assert(previousBasepri == 0); // there should be no priority set before entering pendSv
154 
155  uint64_t result = 0;
156 
157  if (s_previousTask != nullptr)
158  {
159  s_previousTask->m_stackPointer = psp;
160  Hooks::taskStopped(*s_previousTask);
161  }
162 
163  if (s_idling)
164  s_idle->m_stackPointer = psp;
165 
166  if (s_nextTask == nullptr)
167  {
168  s_idling = true;
169  s_previousTask = nullptr;
170  result = reinterpret_cast<uint64_t>(s_idle->m_stackPointer);
171  Hooks::enterIdle();
172  }
173  else
174  {
175  s_idling = false;
176  s_previousTask = s_nextTask;
177  s_currentTask = s_nextTask;
178  result = reinterpret_cast<uint64_t>(s_currentTask->m_stackPointer);
179  s_currentTask->m_lastStarted = s_ticks;
180  s_nextTask = nullptr;
181 
182  if(s_currentTask->m_mutex != nullptr) // there is a mutex we need to re-acquire before exit
183  {
184  result |= static_cast<uint64_t>(s_currentTask->m_mutex->reLockFromPendSv(Mutex::CriticalSection(true))) <<32;
185  s_criticalSection = true;
186  s_currentTask->m_mutex = nullptr;
187  Hooks::mutexRestoredForTask(*s_currentTask);
188  }
189 
190  assert(s_currentTask->isStarted());
191  Hooks::taskStarted(*s_currentTask);
192  }
193  return result;
194 }

196 void Scheduler::wakeUp(TaskControlBlock& task, ConditionVariable& condition)
197 {
198  auto previous = CortexM::setBasepri(kServiceCallPriority); // get a lock up to service call
199  assert(task.m_waiting == &condition); // check the condition is the one the task was waiting for
200 
201  condition.removeWaiting(task);
202  task.m_waiting = nullptr;
203  task.setReturnValue(static_cast<uint32_t>(std::cv_status::no_timeout)); // set the return value to no timeout
204 
205  if(task.m_waitUntil.has_value()) // task was also waiting for a timeout
206  {
207  task.m_waitUntil = std::nullopt;
208  s_timeouts.erase(task);
209  }
210 
211  s_ready.insertWhen(TaskControlBlock::priorityIsLower, task);
212  doSwitch(); // ask for a switch if needed (released a task with higher priority)
213  CortexM::setBasepri(previous); // and restore the basepri to its previous value
214 }

216 void Scheduler::updatePriority(TaskControlBlock& task, Priority newPriority)
217 {
218  auto previous = CortexM::setBasepri(kServiceCallPriority); // get a lock up to service call
219 
220  task.m_priority = newPriority;
221  if(task.isStarted()) // check task is started
222  {
223  if(&task == s_currentTask || &task == s_nextTask) // task is current or next to switch to
224  doSwitch(); // ask for a switch, this will compare priority again with other ready tasks
225  else if(task.m_waiting != nullptr) // task is waiting a condition variable
226  {
227  task.m_waiting->removeWaiting(task); // remove
228  task.m_waiting->addWaiting(task); // then re-insert back the task in the list
229  }
230  else if(!task.m_waitUntil.has_value()) // task is not current or next, is not waiting for a condition variable, and not waiting for a timeout, then it is in ready list
231  {
232  s_ready.erase(task); // remove the task from ready list
233  s_ready.insertWhen(TaskControlBlock::priorityIsLower, task); // then re insert it, this will update its order
234  if(&s_ready.front() == &task) // this brought the task to the first place in the ready list
235  doSwitch(); // so test if it is higher priority than current / next task
236  }
237  }
238 
239  //Hooks::taskPriorityChanged(task);
240  CortexM::setBasepri(previous);
241 }

243 void Scheduler::serviceCallHandler(StackFrame* frame,
244  ServiceCallNumber parameter, [[maybe_unused]] bool isThread)
245 {
246  Hooks::enterServiceCall();
247  bool taskSwitch = false;
248 
249  switch (parameter)
250  {
251  default:
252  assert(false);
253  break;
254 
255  case ServiceCallNumber::Terminate:
256  {
257  assert(frame->r0 != 0); // task pointer is nullptr
258  auto& task = *reinterpret_cast<TaskControlBlock*>(frame->r0);
259 
260  if(!task.m_active.exchange(false)) // was not active, abort termination
261  break;
262 
263  s_allTasks.erase(task);
264 
265  if(task.m_waitUntil.has_value())
266  {
267  s_timeouts.erase(task);
268  task.m_waitUntil = std::nullopt;
269  }
270 
271  if(task.m_waiting != nullptr)
272  {
273  task.m_waiting->removeWaiting(task);
274  task.m_waiting = nullptr;
275  }
276 
277  if (&task == s_currentTask)
278  {
279  assert(!s_criticalSection);
280  s_previousTask = s_currentTask = nullptr;
281  taskSwitch = doSwitch();
282  }
283  Hooks::taskTerminated(task);
284  break;
285  }
286 
287  case ServiceCallNumber::Sleep:
288  {
289  assert(isThread); // should not be called from non thread mode
290  assert(!s_criticalSection); // should not be called in critical section
291  assert(s_currentTask != nullptr); // cannot be called if there is no current task running
292 
293  auto delta = duration{static_cast<int32_t>(frame->r0) + 1}; // add one because we want to wait at least the required time
294  assert(delta.count() >= 0);
295  s_currentTask->m_waitUntil = s_ticks + delta;
296  s_timeouts.insertWhen(wakeupAfter, *s_currentTask);
297  Hooks::taskSleep(*s_currentTask);
298  s_currentTask = nullptr;
299  taskSwitch = doSwitch();
300  break;
301  }
302 
303  case ServiceCallNumber::Switch:
304  {
305  assert(isThread); // should not be called from non thread mode
306  assert(!s_criticalSection); // should not be called in critical section
307  taskSwitch = doSwitch();
308  break;
309  }
310 
311  case ServiceCallNumber::Wait:
312  {
313  assert(isThread); // should not be called from non thread mode
314  assert(frame->r0 != 0); // cannot be called with undefined condition variable
315  assert(s_currentTask != nullptr); // cannot be called if there is no current task running
316 
317  ConditionVariable* condition = reinterpret_cast<ConditionVariable*>(frame->r0);
318  duration timeout{frame->r1};
319  Mutex* mutex = reinterpret_cast<Mutex*>(frame->r2);
320 
321  if(timeout.count() >= 0)
322  {
323  s_currentTask->m_waitUntil = s_ticks + timeout;
324  s_timeouts.insertWhen(wakeupAfter, *s_currentTask);
325  Hooks::taskWaitTimeout(*s_currentTask, *condition, s_currentTask->m_waitUntil.value());
326  Hooks::conditionVariableStartWaiting(*condition, *s_currentTask, timeout);
327  }
328  else
329  {
330  Hooks::taskWait(*s_currentTask, *condition);
331  Hooks::conditionVariableStartWaiting(*condition, *s_currentTask);
332  }
333 
334  if(mutex != nullptr)
335  {
336  assert(s_criticalSection == true); // should be in critical section
337  s_currentTask->m_mutex = mutex;
338  mutex->releaseFromServiceCall();
339  mutex->m_criticalSection.disable();
340  s_criticalSection = false;
341  Hooks::mutexStoredForTask(*s_currentTask);
342  }
343 
344  condition->addWaiting(*s_currentTask);
345  s_currentTask->m_waiting = condition;
346  s_currentTask = nullptr;
347  taskSwitch = doSwitch();
348  break;
349  }
350 
351  }
352  Hooks::exitServiceCall(taskSwitch);
353 }
