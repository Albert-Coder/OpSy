52 /**
53  * @brief Puts the @c Task to sleep for a given @c duration
54  * @warning This should only be called from @c Task and never from interrupt service routine
55  * @warning Make sure you release all @c Mutex before you call @c sleep_for
56  */
57 void inline sleep_for(duration t)
58 {
59  asm volatile(
60  "mov r0, %[count] \t\n"
61  "svc %[immediate] \t\n"
62  :
63  : [immediate] "I" (Scheduler::ServiceCallNumber::Sleep), [count] "r" (t.count())
64  : "r0");
65 }

67 /**
68  * @brief Puts the @c Task to sleep up to a @c time_point
69  * @warning This should only be called from @c Task and never from interrupt service routine
70  * @warning Make sure you release all @c Mutex before you call @c sleep_for
71  */
72 void inline sleep_until(time_point tp)
73 {
74  using namespace std::chrono_literals;
75  assert(tp-Scheduler::now() < 1h); // if you sleep for more than 1h you probably are missing something (low power mode)
76  sleep_for(tp - Scheduler::now());
77 }
